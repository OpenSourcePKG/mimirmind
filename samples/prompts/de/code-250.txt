Ueberarbeite das folgende Python-Modul so, dass (a) doppelte Dictionary-Zugriffe entfernt werden, (b) der datetime.now()-Aufruf aus der Schleife gehoben wird und (c) Typannotationen ergaenzt werden. Gib das komplette ueberarbeitete Modul zurueck.

def process_0(records, config, sink):
    result = []
    for r in records:
        ts = datetime.now().isoformat()
        if config['flags']['strict'] and not r.get('id'):
            continue
        row = {
            'id':     r.get('id'),
            'ts':     ts,
            'name':   config['naming']['prefix'] + r.get('name', ''),
            'weight': r.get('weight', 0) * config['weights']['scale'],
        }
        if config['flags']['dedupe']:
            if row['id'] in seen:
                continue
            seen.add(row['id'])
        sink.write(row)
        result.append(row)
    return result
